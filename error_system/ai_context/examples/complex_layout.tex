\newcommand{\sidebarheading}[1]{%
  \vspace{0.3cm}
  {\headingfont\large\color{white}#1}\\[0.1cm]
  \textcolor{accent2}{\rule{4cm}{2pt}}
  \vspace{0.2cm}
}

\newcommand{\mainheading}[1]{%
  \vspace{0.3cm}
  \begin{tikzpicture}
    \node[anchor=west, font=\headingfont\Large, text=sidebar] at (0,0) {#1};
    \draw[accent1, line width=3pt] (0,-0.15) -- (3,-0.15);
  \end{tikzpicture}
  \vspace{0.2cm}
}

\newcommand{\experienceentry}[5]{%
  \begin{tikzpicture}
    % Timeline dot
    \fill[accent1] (0,0.2) circle (0.15cm);
    % Content box
    \node[anchor=west, text width=13cm] at (0.5,0) {%
      \textbf{\color{sidebar}#1} \hfill {\color{accent1}#2}\\
      \textit{\color{textlight}#3}\\[0.1cm]
      #4
    };
    % Connection line
    \draw[accent1, line width=2pt] (0,0.2) -- (0.3,0.2);
  \end{tikzpicture}
  \vspace{#5}
}

\newcommand{\skillcircle}[2]{%
  \begin{tikzpicture}
    \draw[gray!20, line width=4pt] (0,0) circle (0.8cm);
    \draw[accent2, line width=4pt] (0,0) arc[start angle=90, end angle=90-#2*3.6, radius=0.8cm];
    \node[font=\small\bfseries] at (0,0) {#2\%};
    \node[below, font=\tiny, text width=1.8cm, align=center] at (0,-1.1) {#1};
  \end{tikzpicture}
}

\newcommand{\timelinedot}[3]{%
  \begin{tikzpicture}
    \fill[#1] (0,0) circle (0.2cm);
    \node[right, font=\small\bfseries] at (0.3,0.1) {#2};
    \node[right, font=\scriptsize\color{textlight}] at (0.3,-0.2) {#3};
  \end{tikzpicture}
}
